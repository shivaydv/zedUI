## Usage

```tsx
import { FAQ } from '@/components/FAQ';
```
```tsx
const faqItems = [
  {
    question: '...',
    answer: '...',
  },
  // Add more FAQ items as needed
];
```

```tsx
<FAQ
  items={faqItems}
  stagger={true}
  delay={0.15}
  animated={true}
/>
```
